def plt_scatter_fke_data(srt,nmbr):
    t = np.linspace(0,np.pi*100,np.int(1e5))
    N = 10+np.sin(t)
    bkgd = stats.norm.rvs(size=np.int(1e5))*np.sqrt(N)+N
    colors = (0,0,0)
    area = np.pi*3
    plt.scatter(t[srt:(srt+nmbr)], bkgd[srt:(srt+nmbr)], s=area, alpha=0.5)
    plt.title('')
    plt.xlabel('Time')
    plt.ylabel('Background')
    plt.show()
